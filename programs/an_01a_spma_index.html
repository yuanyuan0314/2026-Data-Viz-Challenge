Purpose

Compute the Small-Producer Market Access (SPMA) index for every county from distance_SPMA_merged.csv, then map it as a US county choropleth.

The SPMA index

For county \(i\), let

  • \(d^{\text{FM}}_i\) = distance from the county centroid to the nearest farmers market (miles),
  • \(d^{\text{hub}}_i\) = distance from the county centroid to the nearest food hub (miles),

and define the three sales pieces from the county’s total sales all_sales:

\[ \text{DTC}_i = \text{all_sales}_i \times \text{d2c_sales_pct}_i/100,\qquad \text{Int}_i = \text{all_sales}_i \times \text{intermediated_sales_pct}_i/100 . \]

Then

\[ \text{SPMA}_i = \dfrac{\text{DTC}_i}{\left(d^{\text{FM}}_i\right)^2} \;+\; \dfrac{\text{Int}_i}{\left(d^{\text{hub}}_i\right)^2} \]

Step 1 — read the merged county table

dat <- read_csv("../data/outcome/distance_SPMA_merged.csv") %>%
  mutate(county_fips = sprintf("%05d", as.integer(county_fips)))

Step 2 — derive intermediated sales share from DTC share

d2c_sales_pct and all_sales have the fewest missing values of any of the sales variables, so rather than relying on the sparser intermediated_sales_pct column directly, we back it out as the complement of DTC: 1 - d2c_sales_pct. Two observations have d2c_sales_pct>100 (James City, VA and Rockdale, GA) and are assigned NA.

dat <- dat %>%
  mutate(
    intermediated_sales_pct_derived = if_else(
      d2c_sales_pct > 100,
      NA,
      100 - d2c_sales_pct
    )
  )

Step 3 — compute the SPMA index

spma_index <- dat %>%
  mutate(
    
    has_hub  = !is.na(number_foodhub) & number_foodhub > 0,
    dtc_sales          = all_sales * d2c_sales_pct/100,
    interm_sales       = all_sales * intermediated_sales_pct_derived/100,
    SPMA = dtc_sales / (distance_nearest_farmersmarket_mile^2) +
           interm_sales  / (distance_nearest_hub_mile^2)
  )

# quick look
spma_index %>%
  summarise(n = n(),
            n_with_hub = sum(has_hub, na.rm = TRUE),
            spma_na = sum(is.na(SPMA)),
            spma_min = min(SPMA, na.rm = TRUE),
            spma_median = median(SPMA, na.rm = TRUE),
            spma_max = max(SPMA, na.rm = TRUE))

Step 4 — Log & Rescale the index to 0-100

rescale_0_100 <- function(x) {
  (x - min(x, na.rm = TRUE)) / (max(x, na.rm = TRUE) - min(x, na.rm = TRUE)) * 100
}

spma_index$SPMA_log <- log(spma_index$SPMA + 1)   # avoid log(0)
spma_index$SPMA_0_100 <- rescale_0_100(spma_index$SPMA_log)

Step 5 — US county choropleth

# lower-48 + DC county polygons (drop AK/HI/territories)
counties_sf <- counties(cb = TRUE, year = 2023, progress_bar = FALSE) %>%
  filter(!STATEFP %in% c("02", "15", "60", "66", "69", "72", "78")) %>%
  transmute(county_fips = GEOID) %>%
  st_transform(5070)

states_sf <- states(cb = TRUE, year = 2023, progress_bar = FALSE) %>%
  filter(!STATEFP %in% c("02", "15", "60", "66", "69", "72", "78")) %>%
  st_transform(5070)

map_df <- counties_sf %>% left_join(spma_index, by = "county_fips")

p <- ggplot(map_df) +
  geom_sf(aes(fill = SPMA_0_100), color = NA) +
  geom_sf(data = states_sf, fill = NA, color = "grey30", linewidth = 0.2) +
  scale_fill_viridis_c(labels = scales::label_number(), na.value = "grey88",
                       name = "SPMA(0-100)") +
  labs(title = "Small-Producer Market Access (SPMA) index by county",
       subtitle = "Higher = better weighted access to farmers markets & food hubs") +
  theme_void(base_size = 12) +
  theme(plot.background  = element_rect(fill = "white", color = NA),
        panel.background = element_rect(fill = "white", color = NA),
        plot.title    = element_text(face = "bold"),
        plot.subtitle = element_text(color = "grey35"),
        plot.caption  = element_text(color = "grey55"),
        legend.position = c(0.92, 0.25))
p
